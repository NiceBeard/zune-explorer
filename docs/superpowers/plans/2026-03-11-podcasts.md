# Podcast Support Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add podcast subscription, streaming, and download support as a new top-level category in Zune Explorer.

**Architecture:** Modular approach — `src/main/podcast-manager.js` handles RSS parsing, downloads, and persistence in the main process. `src/assets/js/podcast-renderer.js` provides `PodcastPanel` class for all UI rendering. Main `renderer.js` delegates to PodcastPanel when the podcasts category is selected.

**Tech Stack:** Electron IPC, `fast-xml-parser` (RSS/Atom/OPML), iTunes Search API, Node `https` streams, VirtualScroller, AudioPlayer.

**Spec:** `docs/superpowers/specs/2026-03-11-podcasts-design.md`

Create `src/main/podcast-manager.js` with:

- `_loadSubscriptions()` / `_saveSubscriptions()` — read/write `userData/podcasts/subscriptions.json`

- `_loadPreferences()` / `_savePreferences()` — `userData/podcasts/preferences.json`
- `getSubscriptions()` — returns subscription list (no episodes)
- `getEpisodes(subscriptionId)` — returns episodes for one subscription
- `getPreferences()` — returns preferences object

```javascript
const fs = require('fs');
const path = require('path');

class PodcastManager {
  constructor(userDataPath) {
    this.dataDir = path.join(userDataPath, 'podcasts');
    this.episodesDir = path.join(this.dataDir, 'episodes');
    this.artworkDir = path.join(this.dataDir, 'artwork');
    this.subscriptionsPath = path.join(this.dataDir, 'subscriptions.json');
    this.preferencesPath = path.join(this.dataDir, 'preferences.json');

    // Ensure directories exist
    for (const dir of [this.dataDir, this.episodesDir, this.artworkDir]) {
      if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
    }

    this.subscriptions = this._loadSubscriptions();
    this.preferences = this._loadPreferences();
    this._downloadQueue = [];
    this._activeDownloads = new Map();
    this._lastPlaybackState = null; // for before-quit persistence
  }

  // --- Persistence ---

  _loadSubscriptions() {
    try {
      return JSON.parse(fs.readFileSync(this.subscriptionsPath, 'utf-8'));
    } catch { return []; }
  }

  _saveSubscriptions() {
    fs.writeFileSync(this.subscriptionsPath, JSON.stringify(this.subscriptions, null, 2));
  }

  _loadEpisodes(subscriptionId) {
    try {
      const p = path.join(this.episodesDir, `${subscriptionId}.json`);
      return JSON.parse(fs.readFileSync(p, 'utf-8'));
    } catch { return []; }
  }

  _saveEpisodes(subscriptionId, episodes) {
    const p = path.join(this.episodesDir, `${subscriptionId}.json`);
    fs.writeFileSync(p, JSON.stringify(episodes, null, 2));
  }

  _loadPreferences() {
    try {
      return JSON.parse(fs.readFileSync(this.preferencesPath, 'utf-8'));
    } catch { return {}; }
  }

  _savePreferences() {
    fs.writeFileSync(this.preferencesPath, JSON.stringify(this.preferences, null, 2));
  }

  // --- Public getters ---

  getSubscriptions() {
    return this.subscriptions;
  }

  getEpisodes(subscriptionId) {
    return this._loadEpisodes(subscriptionId);
  }

  getPreferences() {
    return this.preferences;
  }
}

module.exports = PodcastManager;
```

```javascript
// --- Feed fetching ---

_fetchUrl(url, _redirectCount = 0) {
  return new Promise((resolve, reject) => {
    if (_redirectCount > 5) return reject(new Error('Too many redirects'));
    const client = url.startsWith('https') ? https : http;
    const req = client.get(url, { headers: { 'User-Agent': 'ZuneExplorer/1.4.0' } }, (res) => {
      // Follow redirects (up to 5)
      if ([301, 302, 307, 308].includes(res.statusCode) && res.headers.location) {
        return this._fetchUrl(res.headers.location, _redirectCount + 1).then(resolve, reject);
      }
      if (res.statusCode === 401 || res.statusCode === 403) {
        return reject(new Error('This feed requires authentication'));
      }
      if (res.statusCode !== 200) {
        return reject(new Error(`HTTP ${res.statusCode}`));
      }
      const chunks = [];
      res.on('data', (chunk) => chunks.push(chunk));
      res.on('end', () => resolve(Buffer.concat(chunks)));
      res.on('error', reject);
    });
    req.on('error', reject);
    req.setTimeout(15000, () => { req.destroy(); reject(new Error('Request timed out')); });
  });
}

_parseFeed(xml) {
  const parsed = this._xmlParser.parse(xml);
  // RSS 2.0
  const channel = parsed?.rss?.channel;
  if (channel) return this._parseRSSChannel(channel);
  // Atom
  const feed = parsed?.feed;
  if (feed) return this._parseAtomFeed(feed);
  throw new Error("Feed isn't valid RSS");
}

_parseRSSChannel(channel) {
  const items = Array.isArray(channel.item) ? channel.item : (channel.item ? [channel.item] : []);
  const artworkUrl = channel['itunes:image']?.['@_href']
    || channel.image?.url
    || null;

  return {
    title: channel.title || 'Untitled Podcast',
    author: channel['itunes:author'] || channel.managingEditor || '',
    description: channel.description || '',
    artworkUrl,
    category: this._detectCategory(channel),
    episodes: items.map((item) => this._parseRSSItem(item)),
  };
}

_parseRSSItem(item) {
  const enclosure = item.enclosure || {};
  const guid = item.guid?.['#text'] || item.guid || null;
  const enclosureUrl = enclosure['@_url'] || null;
  const id = guid || enclosureUrl || crypto.randomUUID();

  return {
    id,
    title: item.title || 'Untitled Episode',
    description: item['itunes:summary'] || item.description || '',
    publishDate: item.pubDate ? new Date(item.pubDate).toISOString() : null,
    duration: this._parseDuration(item['itunes:duration']),
    enclosureUrl,
    enclosureType: enclosure['@_type'] || 'audio/mpeg',
    enclosureSize: parseInt(enclosure['@_length']) || 0,
    played: false,
    playbackPosition: 0,
    downloaded: false,
    localPath: null,
  };
}

_parseAtomFeed(feed) {
  const entries = Array.isArray(feed.entry) ? feed.entry : (feed.entry ? [feed.entry] : []);
  return {
    title: feed.title?.['#text'] || feed.title || 'Untitled Podcast',
    author: feed.author?.name || '',
    description: feed.subtitle || '',
    artworkUrl: feed.icon || feed.logo || null,
    category: 'audio',
    episodes: entries.map((entry) => {
      const link = Array.isArray(entry.link)
        ? entry.link.find(l => l['@_rel'] === 'enclosure') || entry.link[0]
        : entry.link;
      const enclosureUrl = link?.['@_href'] || null;
      const id = entry.id || enclosureUrl || crypto.randomUUID();
      return {
        id,
        title: entry.title?.['#text'] || entry.title || 'Untitled',
        description: entry.summary?.['#text'] || entry.summary || '',
        publishDate: entry.published || entry.updated || null,
        duration: this._parseDuration(entry['itunes:duration']),
        enclosureUrl,
        enclosureType: link?.['@_type'] || 'audio/mpeg',
        enclosureSize: parseInt(link?.['@_length']) || 0,
        played: false,
        playbackPosition: 0,
        downloaded: false,
        localPath: null,
      };
    }),
  };
}

_detectCategory(channel) {
  const cat = channel['itunes:category']?.['@_text'] || '';
  if (/video/i.test(cat) || /video/i.test(channel['itunes:type'] || '')) return 'video';
  // Check if most enclosures are video types
  const items = Array.isArray(channel.item) ? channel.item : [];
  const videoCount = items.filter(i => /video/i.test(i.enclosure?.['@_type'] || '')).length;
  if (videoCount > items.length / 2) return 'video';
  return 'audio';
}

_parseDuration(raw) {
  if (!raw) return 0;
  if (typeof raw === 'number') return raw;
  const str = String(raw);
  const parts = str.split(':').map(Number);
  if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
  if (parts.length === 2) return parts[0] * 60 + parts[1];
  return parseInt(str) || 0;
}
```

```javascript
async subscribe(feedUrl) {
  const xml = await this._fetchUrl(feedUrl);
  const feed = this._parseFeed(xml.toString('utf-8'));
  const id = crypto.randomUUID();

  // Download artwork
  let artworkPath = null;
  if (feed.artworkUrl) {
    try {
      artworkPath = await this._fetchArtwork(id, feed.artworkUrl);
    } catch (e) {
      console.error('Artwork fetch failed:', e.message);
    }
  }

  const subscription = {
    id,
    feedUrl,
    title: feed.title,
    author: feed.author,
    description: feed.description,
    artworkUrl: feed.artworkUrl,
    artworkPath,
    category: feed.category,
    episodeCount: feed.episodes.length,
    newEpisodeCount: feed.episodes.length,
    lastRefreshed: new Date().toISOString(),
    subscribedAt: new Date().toISOString(),
    error: null,
  };

  this.subscriptions.push(subscription);
  this._saveSubscriptions();
  this._saveEpisodes(id, feed.episodes);

  return subscription;
}

async _fetchArtwork(subscriptionId, url) {
  const data = await this._fetchUrl(url);
  const ext = url.match(/\.(jpe?g|png|webp)/i)?.[1] || 'jpg';
  const filename = `${subscriptionId}.${ext}`;
  const filepath = path.join(this.artworkDir, filename);
  fs.writeFileSync(filepath, data);
  return `artwork/${filename}`;
}
```

```javascript
async unsubscribe(subscriptionId) {
  this.subscriptions = this.subscriptions.filter(s => s.id !== subscriptionId);
  this._saveSubscriptions();
  // Clean up episode file
  const epPath = path.join(this.episodesDir, `${subscriptionId}.json`);
```

```javascript
async refresh(subscriptionId) {
  const sub = this.subscriptions.find(s => s.id === subscriptionId);
  if (!sub) throw new Error('Subscription not found');

  try {
    const xml = await this._fetchUrl(sub.feedUrl);
    const feed = this._parseFeed(xml.toString('utf-8'));
    const existingEpisodes = this._loadEpisodes(subscriptionId);
    const existingIds = new Set(existingEpisodes.map(e => e.id));

    // Find new episodes
    const newEpisodes = feed.episodes.filter(e => !existingIds.has(e.id));
    const merged = [...newEpisodes, ...existingEpisodes];

    this._saveEpisodes(subscriptionId, merged);

    // Update subscription metadata
    sub.title = feed.title;
    sub.author = feed.author;
    sub.episodeCount = merged.length;
    sub.newEpisodeCount = newEpisodes.length;
    sub.lastRefreshed = new Date().toISOString();
    sub.error = null;

    // Re-fetch artwork if missing
    if (!sub.artworkPath && feed.artworkUrl) {
      try {
        sub.artworkPath = await this._fetchArtwork(subscriptionId, feed.artworkUrl);
      } catch {}
    }

    this._saveSubscriptions();
    return { subscriptionId, newEpisodeCount: newEpisodes.length };
  } catch (e) {
    sub.error = e.message;
    sub.lastRefreshed = new Date().toISOString();
    this._saveSubscriptions();
    throw e;
  }
}

async refreshAll(webContents) {
  const results = [];
  const batchSize = 5;
  for (let i = 0; i < this.subscriptions.length; i += batchSize) {
    const batch = this.subscriptions.slice(i, i + batchSize);
    const batchResults = await Promise.allSettled(
      batch.map(async (sub) => {
        const result = await this.refresh(sub.id);
        if (webContents) {
          webContents.send('podcast-refresh-complete', result);
        }
        return result;
      })
    );
    results.push(...batchResults);
  }
  return results;
}
```

```javascript
async search(query) {
  const encoded = encodeURIComponent(query);
  const url = `https://itunes.apple.com/search?media=podcast&term=${encoded}&limit=20`;
  const data = await this._fetchUrl(url);
  const json = JSON.parse(data.toString('utf-8'));

  // Fetch artwork thumbnails in parallel (spec: main process proxies all artwork)
  const results = (json.results || []).map(r => ({
    trackName: r.trackName || r.collectionName,
    artistName: r.artistName,
    artworkUrl100: r.artworkUrl100,
    artworkBase64: null,
    feedUrl: r.feedUrl,
    trackCount: r.trackCount,
    genres: r.genres,
  }));

  // Fetch artwork as base64 for CSP compliance (img-src does not allow https:)
  await Promise.allSettled(results.map(async (r) => {
    if (!r.artworkUrl100) return;
    try {
      const imgData = await this._fetchUrl(r.artworkUrl100);
      r.artworkBase64 = `data:image/jpeg;base64,${imgData.toString('base64')}`;
    } catch {}
  }));

  return results;
}
```

```javascript
parseOPML(xml) {
  const parsed = this._xmlParser.parse(xml);
  const body = parsed?.opml?.body;
  if (!body) throw new Error('Invalid OPML file');

  const feeds = [];
  const extract = (outline) => {
    if (!outline) return;
    const items = Array.isArray(outline) ? outline : [outline];
    for (const item of items) {
      if (item['@_xmlUrl']) {
        feeds.push({
          feedUrl: item['@_xmlUrl'],
          title: item['@_title'] || item['@_text'] || '',
        });
      }
      if (item.outline) extract(item.outline);
    }
  };
  extract(body.outline);
  return feeds;
}

async importOPML(filePath, webContents) {
  const xml = fs.readFileSync(filePath, 'utf-8');
  const feeds = this.parseOPML(xml);
  let imported = 0;
  let processed = 0;

  const batchSize = 5;
  for (let i = 0; i < feeds.length; i += batchSize) {
    const batch = feeds.slice(i, i + batchSize);
    await Promise.allSettled(
      batch.map(async (feed) => {
        try {
          // Skip if already subscribed
          if (!this.subscriptions.some(s => s.feedUrl === feed.feedUrl)) {
            await this.subscribe(feed.feedUrl);
            imported++;
          }
        } catch (e) {
          console.error(`OPML import failed for ${feed.feedUrl}:`, e.message);
        }
        processed++;
        if (webContents) {
          webContents.send('podcast-import-progress', {
            current: processed,
            total: feeds.length,
            title: feed.title,
          });
        }
      })
    );
  }
  return { imported, total: feeds.length };
}
```

```javascript
_sanitizeFilename(name) {
  return name.replace(/[<>:"/\\|?*\x00-\x1f]/g, '_').substring(0, 200);
}

async pickDownloadDirectory(dialog) {
  const result = await dialog.showOpenDialog({
    properties: ['openDirectory', 'createDirectory'],
    title: 'Choose podcast download folder',
  });
  if (result.canceled || !result.filePaths.length) return null;
  this.preferences.downloadDirectory = result.filePaths[0];
  this._savePreferences();
  return this.preferences.downloadDirectory;
}

async downloadEpisode(subscriptionId, episodeId, webContents) {
  const sub = this.subscriptions.find(s => s.id === subscriptionId);
  if (!sub) throw new Error('Subscription not found');
  const episodes = this._loadEpisodes(subscriptionId);
  const episode = episodes.find(e => e.id === episodeId);
  if (!episode) throw new Error('Episode not found');
  if (!episode.enclosureUrl) throw new Error('No download URL');

  const downloadDir = this.preferences.downloadDirectory;
  if (!downloadDir) throw new Error('No download directory set');

  // Create podcast subdirectory
  const podcastDir = path.join(downloadDir, this._sanitizeFilename(sub.title));
  if (!fs.existsSync(podcastDir)) fs.mkdirSync(podcastDir, { recursive: true });

  // Determine filename from enclosure URL or episode title
  const urlExt = episode.enclosureUrl.match(/\.(\w{2,4})(\?|$)/)?.[1] || 'mp3';
  const filename = `${this._sanitizeFilename(episode.title)}.${urlExt}`;
  const finalPath = path.join(podcastDir, filename);
  const partialPath = finalPath + '.partial';

  return new Promise((resolve, reject) => {
    const client = episode.enclosureUrl.startsWith('https') ? https : http;
    const req = client.get(episode.enclosureUrl, {
      headers: { 'User-Agent': 'ZuneExplorer/1.4.0' },
    }, (res) => {
      if (res.statusCode !== 200) {
        return reject(new Error(`HTTP ${res.statusCode}`));
      }

      const totalBytes = parseInt(res.headers['content-length']) || episode.enclosureSize || 0;
      let downloadedBytes = 0;
      const writeStream = fs.createWriteStream(partialPath);

      this._activeDownloads.set(episodeId, { req, writeStream, partialPath });

      res.on('data', (chunk) => {
        downloadedBytes += chunk.length;
        if (webContents) {
          webContents.send('podcast-download-progress', {
            episodeId,
            percent: totalBytes > 0 ? Math.round((downloadedBytes / totalBytes) * 100) : 0,
            bytesDownloaded: downloadedBytes,
            bytesTotal: totalBytes,
          });
        }
      });

      res.pipe(writeStream);

      writeStream.on('finish', () => {
        this._activeDownloads.delete(episodeId);
        // Rename partial to final
        fs.renameSync(partialPath, finalPath);
        // Update episode record
        episode.downloaded = true;
        episode.localPath = finalPath;
        this._saveEpisodes(subscriptionId, episodes);

        if (webContents) {
          webContents.send('podcast-download-complete', { episodeId, localPath: finalPath });
        }
        resolve({ episodeId, localPath: finalPath });
      });

      writeStream.on('error', (err) => {
        this._activeDownloads.delete(episodeId);
        try { fs.unlinkSync(partialPath); } catch {}
        if (webContents) {
          webContents.send('podcast-download-error', { episodeId, error: err.message });
        }
        reject(err);
      });
    });

    req.on('error', (err) => {
      this._activeDownloads.delete(episodeId);
      try { fs.unlinkSync(partialPath); } catch {}
      if (webContents) {
        webContents.send('podcast-download-error', { episodeId, error: err.message });
      }
      reject(err);
    });

    req.setTimeout(30000, () => {
      req.destroy();
      reject(new Error('Download timed out'));
    });
  });
}

cancelDownload(episodeId) {
  const active = this._activeDownloads.get(episodeId);
  if (active) {
    active.req.destroy();
    try { active.writeStream.destroy(); } catch {}
    try { fs.unlinkSync(active.partialPath); } catch {}
    this._activeDownloads.delete(episodeId);
  }
}

deleteDownload(subscriptionId, episodeId) {
  const episodes = this._loadEpisodes(subscriptionId);
  const episode = episodes.find(e => e.id === episodeId);
  if (episode && episode.localPath) {
    try { fs.unlinkSync(episode.localPath); } catch {}
    episode.downloaded = false;
    episode.localPath = null;
    this._saveEpisodes(subscriptionId, episodes);
  }
}
```

```javascript
savePlaybackPosition(subscriptionId, episodeId, position) {
  const episodes = this._loadEpisodes(subscriptionId);
  const episode = episodes.find(e => e.id === episodeId);
  if (episode) {
    episode.playbackPosition = position;
    this._saveEpisodes(subscriptionId, episodes);
  }
  // Store for before-quit fast persistence
  this._lastPlaybackState = { subscriptionId, episodeId, position };
}

markPlayed(subscriptionId, episodeId, played) {
  const episodes = this._loadEpisodes(subscriptionId);
  const episode = episodes.find(e => e.id === episodeId);
  if (episode) {
    episode.played = played;
    if (played) episode.playbackPosition = 0;
    this._saveEpisodes(subscriptionId, episodes);
  }
  // Update subscription new count
  const sub = this.subscriptions.find(s => s.id === subscriptionId);
  if (sub) {
    sub.newEpisodeCount = episodes.filter(e => !e.played).length;
    this._saveSubscriptions();
  }
}

persistLastPlaybackState() {
  if (this._lastPlaybackState) {
    const { subscriptionId, episodeId, position } = this._lastPlaybackState;
```

```javascript
    progressHtml = ` &middot; <span class="podcast-ep-remaining">${this._formatDuration(remaining)} left</span>`;
    progressHtml += `<div class="podcast-ep-progress"><div class="podcast-ep-progress-bar" style="width: ${pct}%"></div></div>`;
  }

  // Action buttons
  let actionHtml = '';
  if (episode.downloaded) {
    actionHtml = `<span class="podcast-ep-action downloaded ${played ? 'dim' : ''}">&#10003; downloaded</span>`;
  } else {
    actionHtml = `<span class="podcast-ep-action stream" data-action="stream">&#9654; stream</span>`;
    actionHtml += `<span class="podcast-ep-action download" data-action="download">&#8615; download</span>`;
  }

  // Publish date
  const pubDate = episode.publishDate ? new Date(episode.publishDate).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' }) : '';

  row.innerHTML = `
    ${dotHtml}
    <div class="podcast-ep-info ${played ? 'played' : ''}">
      <div class="podcast-ep-title">${this._escapeHtml(episode.title)}</div>
      <div class="podcast-ep-meta">${pubDate} &middot; ${duration}${progressHtml}</div>
    </div>
    <div class="podcast-ep-actions">${actionHtml}</div>
  `;

  // Event: click title to play
  const titleEl = row.querySelector('.podcast-ep-title');
  if (titleEl) {
    titleEl.addEventListener('click', () => this._playEpisode(episode));
  }

  // Event: stream button
  const streamBtn = row.querySelector('[data-action="stream"]');
  if (streamBtn) {
    streamBtn.addEventListener('click', (e) => { e.stopPropagation(); this._playEpisode(episode); });
  }

  // Event: download button
  const downloadBtn = row.querySelector('[data-action="download"]');
  if (downloadBtn) {
    downloadBtn.addEventListener('click', (e) => { e.stopPropagation(); this._downloadEpisode(episode); });
  }

  // Context menu
  row.addEventListener('contextmenu', (e) => {
    e.preventDefault();
    this._showEpisodeContextMenu(e, episode);
  });

  return row;
}

_formatDuration(seconds) {
  if (!seconds || seconds <= 0) return '';
  const h = Math.floor(seconds / 3600);
  const m = Math.floor((seconds % 3600) / 60);
  if (h > 0) return `${h}h ${m.toString().padStart(2, '0')}m`;
  return `${m}m`;
}
```

```javascript
_registerEventListeners() {
  // Guard: only register once (called from render(), not per-view)
  if (this._listenersRegistered) return;
  this._listenersRegistered = true;

  // Playback position saving (throttled)
  this._timeupdateHandler = (data) => {
    const currentFile = this.explorer.audioPlayer.getCurrentFile();
    if (!currentFile || !currentFile.isPodcast) return;

    const now = Date.now();
    if (now - this._lastPositionSave >= this._positionThrottleMs) {
      this._lastPositionSave = now;
      window.electronAPI.podcastSavePlaybackPosition(
        currentFile.subscriptionId,
        currentFile.id,
        data.currentTime
      );
    }
  };
  this.explorer.audioPlayer.on('timeupdate', this._timeupdateHandler);

  // Pause → save position immediately
  this._pauseHandler = () => {
    const currentFile = this.explorer.audioPlayer.getCurrentFile();
    if (!currentFile || !currentFile.isPodcast) return;
    window.electronAPI.podcastSavePlaybackPosition(
      currentFile.subscriptionId,
      currentFile.id,
      this.explorer.audioPlayer.audio.currentTime
    );
  };
  this.explorer.audioPlayer.on('pause', this._pauseHandler);

  // Track change → detect episode completion
  this._trackchangeHandler = ({ file }) => {
    // Check if previous track was a podcast that just ended
    const prev = this._lastPlayingPodcast;
    if (prev && prev.isPodcast && (!file || file.id !== prev.id)) {
      // Previous podcast episode ended
      window.electronAPI.podcastMarkPlayed(prev.subscriptionId, prev.id, true);
    }
    this._lastPlayingPodcast = (file && file.isPodcast) ? file : null;
  };
  this.explorer.audioPlayer.on('trackchange', this._trackchangeHandler);

  // Download events
  this._dlProgressHandler = window.electronAPI.onPodcastDownloadProgress((data) => {
    this._updateDownloadProgress(data);
  });
  this._dlCompleteHandler = window.electronAPI.onPodcastDownloadComplete((data) => {
    this._onDownloadComplete(data);
  });
  this._dlErrorHandler = window.electronAPI.onPodcastDownloadError((data) => {
    this.explorer.showToast(`Download failed: ${data.error}`);
  });
}

_updateDownloadProgress(data) {
  const row = document.querySelector(`[data-episode-id="${data.episodeId}"] .podcast-ep-actions`);
  if (row) {
    row.innerHTML = `<span class="podcast-ep-action downloading">${data.percent}%</span>`;
  }
}

  if (ep) {
    ep.downloaded = true;
    ep.localPath = data.localPath;
  }
  // Re-render episode list
  this._renderEpisodeList();
}
```

```javascript
async _downloadEpisode(episode) {
  const sub = this.currentSubscription;
  // Check if download directory is set
  let prefs = await window.electronAPI.podcastGetPreferences();
  if (!prefs.downloadDirectory) {
    const dir = await window.electronAPI.podcastPickDownloadDirectory();
    if (!dir) return; // cancelled
  }
  try {
    await window.electronAPI.podcastDownloadEpisode(sub.id, episode.id);
  } catch (e) {
    this.explorer.showToast(`Download failed: ${e.message}`);
  }
}

async _refreshSingle(subscriptionId) {
  try {
    await window.electronAPI.podcastRefresh(subscriptionId);
    this.subscriptions = await window.electronAPI.podcastGetSubscriptions();
    if (this.currentSubscription && this.currentSubscription.id === subscriptionId) {
      this.currentSubscription = this.subscriptions.find(s => s.id === subscriptionId);
      this.episodes = await window.electronAPI.podcastGetEpisodes(subscriptionId);
      this._renderEpisodeList();
    }
    this._updateCount();
    this.explorer.showToast('Feed refreshed');
  } catch (e) {
    this.explorer.showToast(`Refresh failed: ${e.message}`);
  }
}

  try {
    await window.electronAPI.podcastRefresh();
    this.subscriptions = await window.electronAPI.podcastGetSubscriptions();
    this._updateCount();
    this.render();
    this.explorer.showToast('All feeds refreshed');
  } catch (e) {
    this.explorer.showToast(`Refresh failed: ${e.message}`);
  }
}

async _unsubscribe(subscriptionId) {
  const sub = this.subscriptions.find(s => s.id === subscriptionId);
  const confirmed = await this.explorer.showConfirmModal(
    'Unsubscribe',
    `Remove "${sub?.title || 'this podcast'}" from your subscriptions?`
  );
  if (!confirmed) return;
  await window.electronAPI.podcastUnsubscribe(subscriptionId);
  this.subscriptions = await window.electronAPI.podcastGetSubscriptions();
  this._updateCount();
  if (this.currentSubscription?.id === subscriptionId) {
    this.currentSubscription = null;
  }
  this.render();
}

  } else {

    items.push({ label: 'Delete download', action: () => this._deleteDownload(episode) });

  });

}

  this.subscriptions = await window.electronAPI.podcastGetSubscriptions();
```
